let allData = []; let filtered = []; let shown = 20; // THIS PART FETCHES THE REAL DATA async function loadRealData() { try { const response = await fetch('data.json'); const data = await response.json(); // Update the date on the hero section document.getElementById('hero-week-date').textContent = data.last_updated; // Set the clips allData = data.clips; filtered = [...allData]; render(); } catch (e) { console.error("Could not load data.json", e); } } function fmt(n) { if (n >= 1e6) return (n / 1e6).toFixed(1) + 'M'; if (n >= 1e3) return Math.round(n / 1e3) + 'K'; return String(n); } function buildRow(c, maxV, delay) { const pct = Math.round((c.views / maxV) * 100); const rankClass = c.rank <= 3 ? ` rank-${c.rank}` : ''; // This makes the row clickable to the actual video URL return `
${c.rank} NEW
${c.title}
${c.streamer} ${c.platform}
${fmt(c.views)}
${pct}%
`; } function render() { const wrap = document.getElementById('rows-wrap'); if (!filtered.length) return; const maxV = filtered[0].views; wrap.innerHTML = filtered.slice(0, shown).map((c, i) => buildRow(c, maxV, i * 30)).join(''); } // Start the loading process loadRealData();
